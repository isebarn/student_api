table program {
  id integer [pk, increment]
  country varchar
  description varchar
  code varchar
  program_price float
  price float
}

table airport {
  id integer [pk, increment]
  name varchar
  code varchar
}

table flight_info {
  id integer [pk, increment]
  airport integer [ref: > airport.id]
  depart_date datetime
  depart_time varchar
  depart_flight_number varchar  
  depart_airline varchar
  return_date datetime
  return_time varchar
  return_flight_number varchar  
  return_airline varchar
}

table host_family_child {
  id integer [pk, increment]
  name varchar
  gender varchar
}

table host_family_pet {
  id integer [pk, increment]
  name varchar
  type varchar
  inside boolean
}

table host_family {
  id integer [pk, increment]
  number integer
  family_name varchar
  father_first_name varchar
  father_last_name varchar 
  father_age integer
  father_occupation varchar
  father_email varchar
  mother_first_name varchar
  mother_last_name varchar 
  mother_age integer
  mother_occupation varchar
  mother_email varchar
  address_line_1 varchar
  address_line_2 varchar
  address_city varchar
  address_postal_code varchar
  address_country varchar
  phone_extension varchar
  phone_number varchar
  host_family_child [ref: < host_family_child.id]
  host_family_pet [ref: < host_family_pet.id]
  smoking boolean
  airport integer [ref: > airport.id]
  profile_link varchar
  school_name varchar
  school_address_line_1 varchar
  school_address_line_2 varchar
  school_address_city varchar
  school_address_postal_code varchar
  school_address_country varchar  
  school_contact varchar
  school_email varchar
}

table account {
  id integer [pk, increment]
  diet float
  region float
  other_reason varchar
  other float
  credits_reason varchar
  credits float
  deposit float
  second_installment float
  third_installment float
  final_installment float
}

table student_personal_data {
  id integer [pk, increment]
  first_name varchar
  last_name varchar
  gender varchar
  email varchar
  program integer [ref: > program.id]
  airport integer [ref: > airport.id]
  host_family integer [ref: > host_family.id]
  account integer [ref: > account.id]
  date_of_application datetime
  date_of_birth datetime
  address_line_1 varchar
  address_line_2 varchar
  address_city varchar
  address_postal_code varchar
  address_country varchar
  phone_extension varchar
  phone_number varchar
  nationality varchar
  school_name varchar
  school_type varchar
  average_grades varchar
  father_first_name varchar
  father_last_name varchar 
  father_address_line_1 varchar
  father_address_line_2 varchar
  father_address_city varchar
  father_address_postal_code varchar
  father_address_country varchar
  mother_first_name varchar
  mother_last_name varchar 
  mother_address_line_1 varchar
  mother_address_line_2 varchar
  mother_address_city varchar
  mother_address_postal_code varchar
