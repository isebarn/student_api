table program {
  id integer [pk, increment]
  country varchar
  description varchar
  code varchar
  program_price float
  price float
}

table airport_code {
  id integer [pk, increment]
  airport varchar
  code varchar
}

table flight_info {
  id integer [pk, increment]
  airport_code integer [ref: > airport_code.id]
  depart_date datetime
  depart_time varchar
  depart_flight_number varchar  
  depart_airline varchar
  return_date datetime
  return_time varchar
  return_flight_number varchar  
  return_airline varchar
}

table host_family {
  id integer [pk, increment]
  number integer
  first_name varchar
  last_name varchar  
  address_line_1 varchar
  address_line_2 varchar
  address_city varchar
  address_postal_code varchar
  address_country varchar
  email varchar
  phone_extension varchar
  phone_number varchar
  airport_code integer [ref: > airport_code.id]
  profile_link varchar  
}

table account {
  id integer [pk, increment]
  program integer [ref: > program.id]
  deposit_prog float
  second_installment float
  third_installment float
  final_installment float
  total_prog float
  credit_1 float
  credit_2 float
  formula varchar
  deposit_paid float
  second_installment_paid float
  third_installment_paid float
  final_installment_paid float
  balance_owed float
}

table student_personal_data {
  id integer [pk, increment]
  first_name varchar
  last_name varchar
  gender varchar
  email varchar
  program integer [ref: > program.id]
  airport_code integer [ref: > airport_code.id]
  date_of_application datetime
  date_of_birth datetime
  address_line_1 varchar
  address_line_2 varchar
  address_city varchar
  address_postal_code varchar
  address_country varchar
  phone_extension varchar
  phone_number varchar
  nationality varchar
  school_name varchar
  school_type varchar
  average_grades varchar
}
