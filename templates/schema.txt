  id integer [pk, increment]
  country varchar
  description varchar
  code varchar
  program_price float
  price float
}

  id integer [pk, increment]

}

table flight_info {
  id integer [pk, increment]
  airport integer [ref: > airport.id]

  depart_time varchar
  depart_flight_number varchar  
  depart_airline varchar
  return_date datetime
  return_time varchar
  return_flight_number varchar  
  return_airline varchar
}
