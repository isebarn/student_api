// embed
table airport {
  id integer [pk, increment]
  name varchar
  code varchar
}

// embed
table flight {
  id integer [pk, increment]
  date datetime
  time varchar
  flight_number varchar  
  airline varchar  
}

// embed
table flight_info {
  id integer [pk, increment]
  airport integer [ref: >> airport.id]
  depart_flight [ref: >> flight.id]
  return_flight [ref: >> flight.id]
}

// embed
table child {
  id integer [pk, increment]
  name varchar
  gender varchar
}

// embed
table pet {
  id integer [pk, increment]
  name varchar
  type varchar
  inside boolean
}

// embed
table host_parent {
  id integer [pk, increment]
  first_name varchar
  last_name varchar 
  age integer
  occupation varchar
  email varchar  
}

// embed
table address {
  id integer [pk, increment]
  line_1 varchar
  line_2 varchar
  city varchar
  postal_code varchar
  country varchar  
}

// embed
table phone {
  id integer [pk, increment]
  extension varchar
  number varchar  
}

// embed
table school {
  id integer [pk, increment]
  name varchar
  contact varchar
  email varchar
  address [ref: >> address.id]
}

// embed
table host_family {
  id integer [pk, increment]
  number varchar
  family_name varchar
  father [ref: >> host_parent.id]
  mother [ref: >> host_parent.id]
  address [ref: >> address.id]
  phone [ref: >> phone.id]
  child [ref: << child.id]
  pet [ref: << pet.id]
  smoking boolean
  airport integer [ref: >> airport.id]
  profile_link varchar
}

// embed
table account {
  id integer [pk, increment]
  diet float
  region float
  other_reason varchar
  other float
  credits_reason varchar
  credits float
  deposit float
  second_installment float
  third_installment float
  final_installment float
}

// embed
table parent {
  id integer [pk, increment]
  first_name varchar
  last_name varchar 
  email varchar
  address [ref: >> address.id] 
}

table program {
  id integer [pk, increment]
  country varchar
  description varchar
  code varchar
  program_price float
  price float
}

table student_profile {
  first_name varchar
  last_name varchar
  age_on_arrival integer
  gender varchar
  email varchar
  length_of_stay varchar
  passport_country varchar
  passport_number varchar
  mother_name varchar
  father_name varchar
  language varchar
  letter varchar
  description varchar
  imagine varchar
  submitted boolean
  interview datetime
  program integer [ref: > program.id]
}

table student_personal_data {
  id integer [pk, increment]
  allergies varchar
  average_grades varchar
  date_of_application datetime
  date_of_birth datetime
  nationality varchar
  school_name varchar
  school_type varchar
  student_profile integer [ref: > student_profile.id]
  account integer [ref: >> account.id]
  address [ref: >> address.id]
  airport integer [ref: >> airport.id]
  host_airport integer [ref: >> airport.id]
  flight_info [ref: >> flight_info.id]
  father [ref: >> parent.id]
  mother [ref: >> parent.id]
  host_family integer [ref: >> host_family.id]
  phone [ref: >> phone.id]
  host_school [ref: >> school.id]
}
